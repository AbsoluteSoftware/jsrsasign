*RSA PRV P5*
SEQ (len=9)
  INT ver (=0)
  INT N
  INT E
  INT D
  INT P
  INT Q
  INT DP
  INT DQ
  INT CO

*DSA PRV P5*
SEQ (len=6)
  INT ver (=0)
  INT P
  INT Q
  INT G
  INT Y (pub)
  INT X (prv)

*ECC PRV P5*
SEQ
  INT ver (=1)
  OCT d (prv)
  [0]
    OID curve
  [1]
    BIT 04xy (pubkey)
