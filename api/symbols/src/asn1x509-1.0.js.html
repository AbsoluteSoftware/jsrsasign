 94 /**
 95  * X.509 Certificate class to sign and generate hex encoded certificate
 96  * @name KJUR.asn1.x509.Certificate
 97  * @class X.509 Certificate class to sign and generate hex encoded certificate
 98  * @param {Array} params JSON object for Certificate parameters
 99  * @extends KJUR.asn1.ASN1Object
100  * @description
101  * <br/>
102  * This class provides Certificate ASN.1 class structure
103  * defined in 
104  * <a href="https://tools.ietf.org/html/rfc5280#section-4.1">
105  * RFC 5280 4.1</a>.
106  * <pre>
107  * Certificate  ::=  SEQUENCE  {
108  *      tbsCertificate       TBSCertificate,
109  *      signatureAlgorithm   AlgorithmIdentifier,
110  *      signatureValue       BIT STRING  }
111  * </pre>
112  * Parameter "params" JSON object can be
113  * the samme as {@link KJUR.asn1.x509.TBSCertificate}. 
114  * Then they are used to generate TBSCertificate.
115  * Additionally just for Certificate, following parameters can be used:
116  * <ul>
117  * <li>{String}sighex - hexadecimal string for 
118  * signatureValue field.</li>
119  * <li>{Object}cakey - RSA/DSA/ECC private key object or 
120  * PEM file which can be passed to {@link KEYUTIL.getKey}.
121  * When "sighex" is not specified and "cakey" is specified,
122  * "sighex" will be automatically set by signing with
123  * "cakey" during getEncodedHex is called.</li>
124  * <li>{TBSCertfificate}tbsobj - 
125  * {@link KJUR.asn1.x509.TBSCertificate} object
126  * instead of specifying TBSCertificate parameters.</li>
127  * </ul>
128  * CAUTION: APIs of this class have been totally updated without
129  * backward compatibility since jsrsasign 9.0.0.<br/>
130  * NOTE1: 'params' can be omitted.<br/>
131  * NOTE2: DSA/ECDSA is also supported for CA signging key from asn1x509 1.0.6.
132  * @example
133  * var cert = new KJUR.asn1x509.Certificate({
134  *  version: 3,
135  *  serial: {hex: "1234..."},
136  *  sigalg: "SHA256withRSAandMGF1",
137  *  ...
138  *  sighex: "1d3f..." // sign() method won't be called
139  * });
140  *
141  * // sighex will by calculated by signing with cakey
142  * var cert = new KJUR.asn1x509.Certificate({
143  *  version: 3,
144  *  serial: {hex: "2345..."},
145  *  sigalg: "SHA256withRSA",
146  *  ...
147  *  cakey: "-----BEGIN PRIVATE KEY..."
148  * });
149  *
150  * // use TBSCertificate object to sign
151  * var cert = new KJUR.asn1x509.Certificate({
152  *  tbsobj: <<OBJ>>,
153  *  sigalg: "SHA256withRSA",
154  *  cakey: "-----BEGIN PRIVATE KEY..."
155  * });
156  */
157 KJUR.asn1.x509.Certificate = function(params) {
158     KJUR.asn1.x509.Certificate.superclass.constructor.call(this);
159     var _KJUR = KJUR,
160 	_KJUR_asn1 = _KJUR.asn1,
161 	_DERBitString = _KJUR_asn1.DERBitString,
162 	_DERSequence = _KJUR_asn1.DERSequence,
163 	_KJUR_asn1_x509 = _KJUR_asn1.x509,
164 	_TBSCertificate = _KJUR_asn1_x509.TBSCertificate,
165 	_AlgorithmIdentifier = _KJUR_asn1_x509.AlgorithmIdentifier;
166 
167     this.params = undefined;
168 
169     /**
170      * set parameter<br/>
171      * @name setByParam
172      * @memberOf KJUR.asn1.x509.Certificate#
173      * @function
174      * @param params {Array} JSON object of certificate parameters
175      * @since jsrsasign 9.0.0 asn1hex 2.0.0
176      * @description
177      * This method will set parameter to this object.
178      * @example
179      * cert = new KJUR.asn1.x509.Certificate();
180      * cert.setByParam({
181      *   version: 3,
182      *   serial: {hex: "1234..."},
183      *   ...
184      * });
185      */
186     this.setByParam = function(params) {
187 	this.params = params;
188     };
189 
190     /**
191      * sign certificate <br/>
192      * @name sign
193      * @memberOf KJUR.asn1.x509.Certificate#
194      * @function
195      * @description
196      * This method signs TBSCertificate with a specified 
197      * private key and algorithm by 
198      * this.params.cakey and this.params.sigalg parameter.
199      * @example
200      * cert = new KJUR.asn1.x509.Certificate({...});
201      * cert.sign()
202      * "-----BEGIN CERTIFICATE-----\r\n..."
203      */
204     this.sign = function() {
205 	var hTBSC = (new _TBSCertificate(this.params)).getEncodedHex();
206 	var sig = new KJUR.crypto.Signature({alg: this.params.sigalg});
207 	sig.init(this.params.cakey);
208 	sig.updateHex(hTBSC);
209 	var sighex = sig.sign();
210 	this.params.sighex = sighex;
211     };
212 
213     /**
214      * get PEM formatted certificate string after signed
215      * @name getPEM
216      * @memberOf KJUR.asn1.x509.Certificate#
217      * @function
218      * @return PEM formatted string of certificate
219      * @since jsrsasign 9.0.0 asn1hex 2.0.0
220      * @description
221      * This method returns a string of PEM formatted 
222      * certificate.
223      * @example
224      * cert = new KJUR.asn1.x509.Certificate({...});
225      * cert.getPEM() →
226      * "-----BEGIN CERTIFICATE-----\r\n..."
227      */
228     this.getPEM = function() {
229 	return hextopem(this.getEncodedHex(), "CERTIFICATE");
230     };
231 
232     this.getEncodedHex = function() {
233 	var params = this.params;
234 	
235 	if (params.tbsobj == undefined) {
236 	    params.tbsobj = new _TBSCertificate(params);
237 	}
238 
239 	if (params.sighex == undefined && params.cakey != undefined) {
240 	    this.sign();
241 	}
242 
243 	if (params.sighex == undefined) {
244 	    throw new Error("sighex or cakey parameter not defined");
245 	}
246 
247 	var a = [];
248 	a.push(params.tbsobj);
249 	a.push(new _AlgorithmIdentifier({name: params.sigalg}));
250 	a.push(new _DERBitString({hex: "00" + params.sighex}));
251 	var seq = new _DERSequence({array: a});
252 	return seq.getEncodedHex();
253     };
254 
255     if (params != undefined) this.params = params;
256 };

259 /**
260  * ASN.1 TBSCertificate structure class
261  * @name KJUR.asn1.x509.TBSCertificate
262  * @class ASN.1 TBSCertificate structure class
263  * @param {Array} params JSON object of TBSCertificate parameters
264  * @extends KJUR.asn1.ASN1Object
265  *
266  * @description
267  * <br/>
268  * NOTE: TBSCertificate class is updated without backward 
269  * compatibility from jsrsasign 9.0.0 asn1x509 2.0.0.
270  * Most of methods are removed and parameters can be set
271  * by JSON object.
272  *
273  * @example
274  * new TBSCertificate({
275  *  version: 3, // this can be omitted, the default is 3.
276  *  serial: {hex: "1234..."}, // DERInteger parameter
277  *  sigalg: "SHA256withRSA",
278  *  issuer: {array:[[{type:'O',value:'Test',ds:'prn'}]]}, // X500Name parameter

284  *  // All extension parameters need to have "extname" parameter additionaly.
285  *  ext:[{ 
286  *   extname:"keyUsage",critical:true,
287  *   names:["digitalSignature","keyEncipherment"]
288  *  },{
289  *   extname:"cRLDistributionPoints",
290  *   array:[{dpname:{full:[{uri:"http://example.com/a1.crl"}]}}]
291  *  }, ...]
292  * })
293  *
294  * var tbsc = new TBSCertificate();
295  * tbsc.setByParam({version:3,serial:{hex:'1234...'},...});
296  */

298     KJUR.asn1.x509.TBSCertificate.superclass.constructor.call(this);
299     var _KJUR = KJUR,
300 	_KJUR_asn1 = _KJUR.asn1,
301 	_KJUR_asn1_x509 = _KJUR_asn1.x509,
302 	_DERTaggedObject = _KJUR_asn1.DERTaggedObject,
303 	_DERInteger = _KJUR_asn1.DERInteger,
304 	_DERSequence = _KJUR_asn1.DERSequence,
305 	_AlgorithmIdentifier = _KJUR_asn1_x509.AlgorithmIdentifier,
306 	_Time = _KJUR_asn1_x509.Time,
307 	_X500Name = _KJUR_asn1_x509.X500Name,
308 	_SubjectPublicKeyInfo = _KJUR_asn1_x509.SubjectPublicKeyInfo;
309

311

313      * get array of ASN.1 object for extensions<br/>
314      * @name setByParam

317      * @param {Array} JSON object of TBSCertificate parameters

326     /**
327      * get ASN.1 object for extensions<br/>
328      * @name getExtDERSequence
329      * @memberOf KJUR.asn1.x509.TBSCertificate#
330      * @function
331      * @param {Array} aExt array of certificate extension parameters (OPTIONAL)
332      * @return {DERSequence} ASN.1 DERSequence of specified extensions
333      * @since jsrsasign 9.0.0 asn1x509 2.0.0
334      * @see KJUR.asn1.DERSequence
335      * @see KJUR.asn1.x509.TBSCertificate
336      * @see KJUR.asn1.x509.Certificate
337      * @see KJUR.asn1.csr.CertificationRequestInfo
338      * @description
339      * This method returns DERSequence object of certificate 
340      * extensions. 
341      * <br/>
342      * When argument "aExt" is not specified,
343      * internal parameter "params.ext" of TBSCertificate 
344      * will be referred.
345      * Specifying "aExt" can be used for
346      * {@link KJUR.asn1.csr.CertificationRequestInfo}.
347      * @example
348      * // get DERSequence of extensions from TBSCertificate object
349      * tbsc = new KJUR.asn1.x509.TBSCertificate({version:3,...,ext:[...]});
350      * tbsc.getExtDERSequence → DERSequence object of extensions
351      * 
352      * // get DERSequence of extensions by the argument
353      * tbsc = new KJUR.asn1.x509.TBSCertificate();
354      * tbsc.getExtDERSequence([{extname:"subjectAltName"...}...]) →
355      * DERSequence by specified array of extension parameters
356      */
357     this.getExtDERSequence = function(aExt) {
358 	if (aExt === undefined &&
359 	    this.params !== null &&
360 	    this.params.ext !== undefined) {
361 	    aExt = this.params.ext;
362 	}
363 	var a = [];
364 	for (var i = 0; i < aExt.length; i++) {
365 	    var param = aExt[i];
366 	    var extname = param.extname;
367 	    var obj = null;
368 
369 	    if (extname == "subjectKeyIdentifier") {
370 		obj = new _KJUR_asn1_x509.SubjectKeyIdentifier(param);
371 	    } else if (extname == "keyUsage") {
372 		obj = new _KJUR_asn1_x509.KeyUsage(param);
373 	    } else if (extname == "subjectAltName") {
374 		obj = new _KJUR_asn1_x509.SubjectAltName(param);
375 	    } else if (extname == "issuerAltName") {
376 		obj = new _KJUR_asn1_x509.IssuerAltName(param);
377 	    } else if (extname == "basicConstraints") {
378 		obj = new _KJUR_asn1_x509.BasicConstraints(param);
379 	    } else if (extname == "cRLDistributionPoints") {
380 		obj = new _KJUR_asn1_x509.CRLDistributionPoints(param);
381 	    } else if (extname == "certificatePolicies") {
382 		obj = new _KJUR_asn1_x509.CertificatePolicies(param);
383 	    } else if (extname == "authorityKeyIdentifier") {
384 		obj = new _KJUR_asn1_x509.AuthorityKeyIdentifier(param);
385 	    } else if (extname == "extKeyUsage") {
386 		obj = new _KJUR_asn1_x509.ExtKeyUsage(param);
387 	    } else if (extname == "authorityInfoAccess") {
388 		obj = new _KJUR_asn1_x509.AuthorityInfoAccess(param);
389 	    } else {
390 		throw new Error("extension not supported:"
391 				+ JSON.stringify(param));
392 	    }
393 	    if (obj != null) a.push(obj);
394 	}
395 
396 	return new _DERSequence({array: a});
397     };
398 
399     this.getEncodedHex = function() {
400 	var a = [];
401 	var params = this.params;
402 
403 	// X.509v3 default if params.version not defined
404 	if (params.version != undefined || params.version != 1) {
405 	    var version = 2; 
406 	    if (params.version != undefined) version = params.version - 1;
407 	    var obj = 
408 		new _DERTaggedObject({obj: new _DERInteger({int: version})}) 
409 	    a.push(obj);
410 	}
411 
412 	a.push(new _DERInteger(params.serial));
413 	a.push(new _AlgorithmIdentifier({name: params.sigalg}));
414 	a.push(new _X500Name(params.issuer));
415 	a.push(new _DERSequence({array:[new _Time({str: params.notbefore}),
416 					new _Time({str: params.notafter})]}));
417 	a.push(new _X500Name(params.subject));
418 	a.push(new _SubjectPublicKeyInfo(KEYUTIL.getKey(params.sbjpubkey)));
419 	if (params.ext !== undefined && params.ext.length > 0) {
420 	    a.push(new _DERTaggedObject({tag: "a3",
421 					 obj: this.getExtDERSequence()}));
422 	}
423 
424 	var seq = new KJUR.asn1.DERSequence({array: a});
425 	return seq.getEncodedHex();
426     };
427 
428     if (params !== undefined) this.setByParam(params);
429 };

436 /**
437  * base Extension ASN.1 structure class
438  * @name KJUR.asn1.x509.Extension
439  * @class base Extension ASN.1 structure class
440  * @param {Array} params associative array of parameters (ex. {'critical': true})
441  * @extends KJUR.asn1.ASN1Object
442  * @description
443  * <pre>
444  * Extension  ::=  SEQUENCE  {
445  *     extnID      OBJECT IDENTIFIER,
446  *     critical    BOOLEAN DEFAULT FALSE,
447  *     extnValue   OCTET STRING  }
448  * </pre>
449  * @example
450  */

452     KJUR.asn1.x509.Extension.superclass.constructor.call(this);
453     var asn1ExtnValue = null,
454 	_KJUR = KJUR,
455 	_KJUR_asn1 = _KJUR.asn1,
456 	_DERObjectIdentifier = _KJUR_asn1.DERObjectIdentifier,
457 	_DEROctetString = _KJUR_asn1.DEROctetString,
458 	_DERBitString = _KJUR_asn1.DERBitString,
459 	_DERBoolean = _KJUR_asn1.DERBoolean,
460 	_DERSequence = _KJUR_asn1.DERSequence;
461 
462     this.getEncodedHex = function() {
463         var asn1Oid = new _DERObjectIdentifier({'oid': this.oid});
464         var asn1EncapExtnValue =
465             new _DEROctetString({'hex': this.getExtnValueHex()});
466 
467         var asn1Array = new Array();
468         asn1Array.push(asn1Oid);
469         if (this.critical) asn1Array.push(new _DERBoolean());
470         asn1Array.push(asn1EncapExtnValue);
471 
472         var asn1Seq = new _DERSequence({'array': asn1Array});
473         return asn1Seq.getEncodedHex();
474     };
475 
476     this.critical = false;
477     if (params !== undefined) {
478         if (params.critical !== undefined) {
479             this.critical = params.critical;
480         }
481     }

485 /**
486  * KeyUsage ASN.1 structure class
487  * @name KJUR.asn1.x509.KeyUsage
488  * @class KeyUsage ASN.1 structure class
489  * @param {Array} params associative array of parameters (ex. {'bin': '11', 'critical': true})
490  * @extends KJUR.asn1.x509.Extension
491  * @description
492  * This class is for <a href="https://tools.ietf.org/html/rfc5280#section-4.2.1.3" target="_blank">KeyUsage</a> X.509v3 extension.
493  * <pre>
494  * id-ce-keyUsage OBJECT IDENTIFIER ::=  { id-ce 15 }
495  * KeyUsage ::= BIT STRING {
496  *   digitalSignature   (0),
497  *   nonRepudiation     (1),
498  *   keyEncipherment    (2),
499  *   dataEncipherment   (3),
500  *   keyAgreement       (4),
501  *   keyCertSign        (5),
502  *   cRLSign            (6),
503  *   encipherOnly       (7),
504  *   decipherOnly       (8) }
505  * </pre><br/>
506  * NOTE: 'names' parameter is supprted since jsrsasign 8.0.14.
507  * @example
508  * o = new KJUR.asn1.x509.KeyUsage({bin: "11"});
509  * o = new KJUR.asn1.x509.KeyUsage({critical: true, bin: "11"});
510  * o = new KJUR.asn1.x509.KeyUsage({names: ['digitalSignature', 'keyAgreement']});
511  */
512 KJUR.asn1.x509.KeyUsage = function(params) {
513     KJUR.asn1.x509.KeyUsage.superclass.constructor.call(this, params);
514     var _KEYUSAGE_NAME = X509.KEYUSAGE_NAME;
515 
516     this.getExtnValueHex = function() {
517         return this.asn1ExtnValue.getEncodedHex();
518     };
519 
520     this.oid = "2.5.29.15";
521     if (params !== undefined) {
522         if (params.bin !== undefined) {
523             this.asn1ExtnValue = new KJUR.asn1.DERBitString(params);
524         }
525 	if (params.names !== undefined &&
526 	    params.names.length !== undefined) {
527 	    var names = params.names;
